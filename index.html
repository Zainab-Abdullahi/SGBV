data = {
    'Age_Group': ['10-14', '15-19', '20-24', '25-29', '30-34', '35-39', '40-44', '45-49', 'Abv 50'],
    'Physical_Violence': [17, 32, 8, 2, 1, 0, 0, 1, 0],
    'Emotional_Violence': [20, 60, 21, 12, 4, 0, 0, 1, 1],
    'Sexual_Violence': [37, 92, 29, 14, 5, 0, 0, 0, 0]
}

df = pd.DataFrame(data)

# Melt the DataFrame to transform it into a long format
df_melted = pd.melt(df, id_vars='Age_Group', var_name='Type_of_Violence', value_name='Number_of_Cases')

# Plotting
plt.figure(figsize=(10, 6))
sns.barplot(x='Age_Group', y='Number_of_Cases', hue='Type_of_Violence', data=df_melted, palette='muted')
plt.title('Violence Cases by Age Group')
plt.xlabel('Age Group')
plt.ylabel('Number of Cases')
plt.xticks(rotation=45)
plt.legend(title='Type of Violence')
plt.show()
